## 文本绘制参数TDrawTextParam
//字体  
DrawFont: TDrawFont  
//文本过长是否换行  
IsWordWrap:Boolean  
//文本显示不下时的截断类型  
FontTrimming:TFontTrimmingType  
fttNone:        不截断  
fttCharacter:   按字母截断  
fftWord:        按单词截断  
//水平对齐风格  
FontHorzAlign:TFontHorzAlign  
fhaLeft:    水平居左  
fhaCenter:  水平居中  
fhaRight:   水平居右  
//垂直对齐风格  
FontVertAlign:TFontVertAlign  
fvaTop:     垂直居上  
fvaCenter:  垂直居中  
fvaBottom:  垂直居下  
//是否垂直绘制  
IsDrawVert:Boolean
